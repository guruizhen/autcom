package de.nulide.shiftcal.logic.io.object;

public class JSONCalendarDate {

    private int year;
    private int month;
    private int day;

    public JSONCalendarDate() {
    }

    public JSONCalendarDate(int year, int month, int day) {
        this.year = year;
        this.month = month;
        this.day = day;
    }

    public int getYear() {
        return year;
    }

    public void setYear(int year) {
        this.year = year;
    }

    public int getMonth() {
        return month;
    }

    public void setMonth(int month) {
        this.month = month;
    }

    public int getDay() {
        return day;
    }

    public void setDay(int day) {
        this.day = day;
    }
}
